#include<stdlib.h>
int tsize;
int hasht(int key)
{
int i ;
i = key%tsize ;
return i;
}
//-------LINEAR PROBING-------
int rehashl(int key)
{
int i ;
i = (key+1)%tsize ;
return i ;
}
//-------QUADRATIC PROBING-------
int rehashq(int key, int j)
{
int i ;
i = (key+(j*j))%tsize ;
return i ;
}

void main()
{

int key,arr[20],hash[20],i,n,s,op,j,k ;

printf ("Enter Elements: ");
for (i=0;i<n;i++)
{
scanf("%d",&arr[i]);

do

switch(op)
{
case 1:
for (i=0;i<tsize;i++)
hash[i]=-1 ;
for(k=0;k<n;k++)
{
key=arr[k] ;

while (hash[i]!=-1)
{
i = rehashl(i);

}

}

case 2:

}
