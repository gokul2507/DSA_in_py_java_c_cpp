#include <stdio.h>
#include<conio.h>
void main()
{
 int array[100], n, c, d, min, t;
 printf("Enter number of elements\n");
 scanf("%d", &n);
 printf("Enter %d integers\n", n);
 for (c = 0; c < n; c++)
 scanf("%d", &array[c]);
 for (c = 0; c < (n - 1); c++) 
 {
 min= c;
 for (d = c + 1; d < n; d++)
 {
 if (array[min] > array[d])
min = d;
 }
 if (min != c)
 {
 t = array[c];
 array[c] = array[min];
 array[min] = t;
 }
 }
 printf("Sorted list in ascending order:\n");
 for (c = 0; c < n; c++)
 printf("%d\n", array[c]);
}
